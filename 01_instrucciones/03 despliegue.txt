## ATENCION: NO USAR CARACTERES ESPECIALES EN LOS NOMBRES DE LOS DIRECTORIOS O ARCHIVOS

## CREAR ENTORNO VIRTUAL
virtualenv .venv

## CREAR ENTORNO VIRTUAL CON VERSION ESPECIFICA
virtualenv --python=/usr/bin/python3.10 .venv

## ACTIVAR ENTORNO VIRTUAL
source .venv/bin/activate

## INSTALAR REQUEREMENTS
pip install -r requirements.txt

## INSTALACION DE .SERVICE Y .SOCKET
sudo cp censoestudiantil.service censoestudiantil.socket /etc/systemd/system

## HABILITAR .SERVICE
sudo systemctl enable censoestudiantil.service

## INICIAR .SOCKET
sudo systemctl start censoestudiantil.socket

## COMPROBAR SOCKET
file /run/censoestudiantil.sock

## INSTALAR BLOQUE DE NGINX
sudo cp censoestudiantil.conf /etc/nginx/sites-available/

## CREAR ENLACE SIMBOLICO
sudo ln -s /etc/nginx/sites-available/censoestudiantil.conf /etc/nginx/sites-enabled/

## REINICIAR NGINX
/etc/init.d/nginx restart

## EDITAR ARCHIVO HOSTS
sudo nano /etc/hosts



## DESCARTAR ERRORES
sudo systemctl status censoestudiantil.socket
sudo systemctl status censoestudiantil.service
sudo systemctl status censoestudiantil


sudo systemctl daemon-reload
sudo systemctl restart censoestudiantil
